1. MLE in the null case

Consider the model \[v | R \sim \mathcal{N}(0, R/N), \quad R | R_0 \sim \mathcal{IW}(\nu_0 \overline{R}, \nu_0 + J + 1), \] for \(\overline{R} = (1 - \lambda) R_0 + \lambda I\). The following code finds the MLE of \(\nu_0\) and \(\lambda\).

maximize_log_pdf <- function(R0, v, N) {
  J <- length(v)
  
  R0 <- (R0 + t(R0)) / 2
  eig <- eigen(R0, symmetric = TRUE)
  
  # Clamp any floating-point negative eigenvalues to exactly 0
  d <- pmax(eig$values, 0) 
  v_tilde <- as.numeric(crossprod(eig$vectors, v))
  
  # Optimize over log_nu0 to keep parameter scales identical
  neg_log_pdf <- function(params) {
    lambda <- params[1]
    nu0 <- exp(params[2]) # Decode log(nu0)
    
    d_lambda <- (1 - lambda) * d + lambda
    
    log_det <- sum(log(d_lambda))
    qf <- N * sum((v_tilde^2) / d_lambda)
    
    inner_log <- 1 + (1 / nu0) * qf
    
    term1 <- lgamma((nu0 + J + 2) / 2)
    term2 <- lgamma((nu0 + 2) / 2)
    term3 <- (J / 2) * log(nu0 * pi)
    term4 <- 0.5 * log_det
    term5 <- ((nu0 + J + 2) / 2) * log(inner_log)
    
    log_lik <- term1 - term2 - term3 - term4 - term5
    
    # Scale objective by J to prevent gradient collapse in finite-differencing
    return(-log_lik / J)
  }
  
  init_lambda <- 0.5
  init_log_nu0 <- log(5.0) 
  
  epsilon <- 1e-5 
  
  # nlminb is significantly more robust for this topology
  opt_res <- nlminb(
    start = c(init_lambda, init_log_nu0),
    objective = neg_log_pdf,
    lower = c(epsilon, log(epsilon)),
    upper = c(1 - epsilon, log(1e5))
  )
  
  return(list(
    lambda = opt_res$par[1],
    nu_0 = exp(opt_res$par[2]),
    # Scale the objective back up by J to get true log likelihood
    max_log_pdf = -opt_res$objective * J,
    convergence = opt_res$convergence, # 0 means successful
    message = opt_res$message
  ))
}

2. MLE in the single effect case (without noise)

Here we implement the MLE of the setting assuming that we know the single true effect and \(N\to \infty\) (so that the variance of \(v\) is 0). This is equivalent to observing the first column of \(R\). The formula appears in Section 2 of the new small writeup.

optimize_params_single_effect <- function(R_col1, R0, init_nu0 = 50, init_lambda = 0.5) {
  
  # Length of the column vector (p dimension)
  p <- length(R_col1)
  
  # Identity matrix of dimension p
  I_p <- diag(p)
  
  # Partition the observed first column of R
  r_11 <- R_col1[1]
  R_21 <- R_col1[-1]
  
  # ---------------------------------------------------------
  # Objective Function (Negative Log-Likelihood)
  # ---------------------------------------------------------
  neg_log_lik <- function(params) {
    nu_0   <- params[1]
    lambda <- params[2]
    
    # 1. Dynamically construct Rbar based on the current lambda
    Rbar <- (1 - lambda) * R0 + lambda * I_p
    
    # 2. Partition the dynamic Rbar matrix
    rbar_11 <- Rbar[1, 1]
    Rbar_21 <- Rbar[-1, 1, drop = FALSE] # Keep as a column matrix
    Rbar_22 <- Rbar[-1, -1, drop = FALSE]
    
    # 3. Compute the Schur Complement of Rbar dynamically
    # Rbar_22.1 = Rbar_22 - Rbar_21 * (1/rbar_11) * Rbar_12
    Rbar_22.1 <- Rbar_22 - (Rbar_21 %*% t(Rbar_21)) / rbar_11
    
    # Compute inverse and log-determinant (safely)
    inv_Rbar_22.1 <- solve(Rbar_22.1)
    log_det_Rbar_22.1 <- as.numeric(determinant(Rbar_22.1, logarithm = TRUE)$modulus)
    
    # --- A. Log-Likelihood of r_11 (Inverse-Gamma) ---
    alpha <- (nu_0 + 2) / 2
    beta  <- (nu_0 * rbar_11) / 2
    
    ll_r11 <- alpha * log(beta) - lgamma(alpha) - 
      (alpha + 1) * log(r_11) - (beta / r_11)
    
    # --- B. Log-Likelihood of R_21 | r_11 (Multivariate t) ---
    mu <- Rbar_21 * (r_11 / rbar_11)
    diff <- R_21 - mu
    
    # Quadratic form: (R_21 - mu)^T %*% Rbar_22.1^{-1} %*% (R_21 - mu)
    Q <- as.numeric(t(diff) %*% inv_Rbar_22.1 %*% diff)
    
    term1 <- lgamma((nu_0 + p + 2) / 2) - lgamma((nu_0 + 3) / 2)
    term2 <- - ((p - 1) / 2) * log(pi)
    term3 <- - ((p - 1) / 2) * log(r_11 / rbar_11)
    term4 <- - 0.5 * log_det_Rbar_22.1
    term5 <- - ((nu_0 + p + 2) / 2) * log(1 + (rbar_11 / r_11) * Q)
    
    ll_R21_given_r11 <- term1 + term2 + term3 + term4 + term5
    
    # Return total negative log-likelihood
    return(-(ll_r11 + ll_R21_given_r11))
  }
  
  # ---------------------------------------------------------
  # Execution of the Optimizer
  # ---------------------------------------------------------
  # "L-BFGS-B" allows us to set both lower and upper bounds
  # nu_0 must be > 0. lambda must be in [0, 1].
  opt_result <- optim(
    par       = c(init_nu0, init_lambda),
    fn        = neg_log_lik,
    method    = "L-BFGS-B",
    lower     = c(1e-4, 1e-4),    # Keep strictly > 0
    upper     = c(Inf, 0.9999),   # lambda caps at 1
    control   = list(maxit = 10000)
  )
  
  return(list(
    nu_0_opt    = opt_result$par[1],
    lambda_opt  = opt_result$par[2],
    neg_log_lik = opt_result$value,
    convergence = opt_result$convergence,
    message     = opt_result$message
  ))
}
